│   ├── main_window.py         # 主窗口 (接受 ctx 参数注入)

│   ├── font_icons.py          # FontIcon 系统 (70+ 图标 + Button/ToggleButton/TextBlock)
│   ├── presenters.py          # Presenter 模板体系 (PresenterRegistry + ContentPresenter)

│   ├── property_panel.py      # 属性编辑面板 (动态表单 + 分组过滤 + 只读模式)

│   ├── result_panel.py        # 结果面板 (历史 FontIcon 状态列 + HTML 帮助)
│   ├── log_panel.py           # 日志面板
│   ├── help_panel.py          # 帮助面板 (节点文档 + 超链接)
│   ├── roi_editor.py          # ROI 编辑器 (全屏模式切换)
│   ├── color_picker.py        # 颜色选择器
│   ├── crop_dialog.py         # 模板裁剪
│   ├── condition_editor.py    # 条件编辑器
│   ├── start_page.py          # 启动页
│   ├── widgets/               # 自定义控件
│   │   ├── grid_splitter_box.py  # GridSplitterBox (90px 阈值)

│   └── node_editor/           # 流程图编辑器
│       ├── scene.py           # 画布场景 (网格/节点/连线 + 撤销重做)
│       ├── node_item.py       # 节点视觉项 (30px 色条 + FontIcon + 主题自适应)

│       └── editor_widget.py   # 编辑器控件 (工具栏 + 小地图 + 拖放)

│   ├── template_matchings/    # 模板匹配 (Base64/SIFT/SURF)
│   ├── detectors/             # 检测 (Blob/Canny/Hough/QR/轮廓)
│   ├── features/              # 特征提取 (AKaze/Brisk/Fast/Freak/Kaze/MSER/Star)
│   ├── others/                # 其他 (拼接/透视变换/Haar/LBP/SVM/HOG/YOLOv3)

    ├── images/                # 测试图像 (33 张)
    ├── icons/                 # 应用图标
    └── videos/                # 测试视频
