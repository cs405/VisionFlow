VisionFlow/
├── main.py                          # 主入口（启动时注册节点+加载插件+显示窗口）
├── requirements.txt                 # Python依赖（PySide6, numpy, opencv-python）
├── settings.yaml                    # 配置文件
├── ui迁移列表.md                     # WPF→PySide6 UI精确移植清单及进度
│
├── core/                            # 核心层（纯逻辑，完全解耦，无Qt依赖）
│   ├── node_base.py                 # 算子抽象基类（Socket/Parameter/Evaluate）
│   ├── workflow.py                  # 流程引擎（节点/连接管理/拓扑排序/序列化）
│   ├── registry.py                  # 节点注册表（单例/自动发现/插件加载）
│   ├── data_packet.py               # 数据包定义（Image/Number/String/Rect等）
│   ├── events.py                    # 事件总线（Pub-Sub, Core→UI单向通知）
│   └── plugin_manager.py            # 插件管理器（预留）
│
├── nodes/                           # 内置算子（纯逻辑，仅依赖core）
│   ├── io/                          # ImageSource, ImageSink
│   ├── preprocessing/               # CvtColor, Resize, GaussianBlur, Threshold
│   ├── feature/                     # Canny, FindContours, FindCircles, FindLines
│   ├── match/                       # TemplateMatch, FeatureMatch
│   └── measurement/                 # Distance, Angle
│
├── gui/                             # UI层（WPF VisionMaster风格, PySide6）
│   ├── theme.py                     # 全局暗色主题：Colors/Fonts/GLOBAL_STYLESHEET
│   ├── title_bar.py                 # WPF CaptionTemplate: 85px标题栏
│   │                                #   Row1: 菜单栏(文件/编辑/运行/系统/帮助)
│   │                                #         + 项目名称 + 系统按钮 + 窗口控制
│   │                                #   Row2: 工具栏(新建/打开/编辑/保存 + 执行)
│   ├── main_window.py              # 主窗口: 无边框 + 三栏QSplitter + 双状态栏
│   │                                #   左: FlowResourcePanel 流程资源
│   │                                #   中: EditorWidget(多Tab) + ResultPanel + 状态
│   │                                #   右: ImageViewer + PropertyPanel
│   │                                #   底部: LogPanel Dock
│   ├── flow_resource_panel.py      # "流程资源"面板: 搜索 + 树形工具箱 + 拖拽
│   ├── node_editor/
│   │   ├── editor_widget.py        # 多Tab流程编辑器: QTabWidget + Zoombox + 连线
│   │   ├── scene.py               # QGraphicsScene: 网格背景 + 节点/连线管理
│   │   ├── node_item.py           # WPF节点: 白底 + 状态条 + 标题栏 + 图标
│   │   ├── socket_item.py         # 端口图形项: 拖拽连线 + 类型着色
│   │   └── edge_item.py           # 连线图形项: 贝塞尔曲线 + 双击删除
│   ├── result_panel.py            # 结果面板: 历史结果/模块结果/帮助 三Tab
│   ├── image_viewer.py            # Zoombox图像查看器: 双击适应/自动缩放/棋盘格
│   ├── property_panel.py          # 属性面板: 动态表单生成 + 事件驱动更新
│   └── log_panel.py               # 日志面板: 级别筛选/导出/暗色主题
│
├── plugins/                         # 插件目录（运行时动态加载）
│   ├── __init__.py
│   └── example_plugin.py            # 示例: 锐化/旋转/翻转/通道分离/直方图均衡化
│
├── projects/                        # 项目保存目录
│
└── utils/                           # 工具模块
    ├── config.py                    # 配置管理
    ├── image_utils.py               # 图像工具函数
    └── logger.py                    # 日志工具
